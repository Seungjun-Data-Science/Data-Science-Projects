Finding Discontinuities¶

In [71]:
discontinuities_one_point_three = []

# Finding discontinuities in years 2008-2017
for year in ['2008','2009','2010','2011','2012','2013','2014','2015','2016','2017']:
    for month in ['01','02','03','04','05','06','07','08','09','10','11','12']:
        date1 = year + "-" + month
        diff1 = approved_df.groupby(['issue_d','sub_grade']).int_rate.mean().to_frame().loc[date1].diff()
        for i in diff1[diff1.int_rate >= 1.3].index.tolist():
            date_leftgrade_rightgrade1 = []
            date_leftgrade_rightgrade1.append(date1)
            date_leftgrade_rightgrade1.append(diff1.iloc[[diff1.index.tolist().index(i)-1]].index[0])
            date_leftgrade_rightgrade1.append(i)
            discontinuities_one_point_three.append(date_leftgrade_rightgrade1)
